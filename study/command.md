# nestjs 글로벌로 인스톨
npm i -g @nestjs/cli

# nestjs 프로젝트 이름으로 생성
nest new project-name

# node 실행
# SWC builder로 더 빠르게 node 실행
npm run start
npm run start -- -b swc

# Lint and autofix with eslint
npm run lint

# Format with prettier
npm run format

# To create a controller using the CLI
nest g controller [name]

# To create a service using the CLI
nest g service [name]
